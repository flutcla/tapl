\section*{Sec1}
\begin{frame}{first frame}
\begin{theorem}
  素数は無限に存在する．
\end{theorem}
\begin{proof}
  \begin{enumerate}
    \item<1-> $p_1, \dots, p_n$ がすべての素数だと仮定する。
    \item<2-> $q:=p_1\dots p_n + 1$ とおく。
    \item<3-> すると $q$ はどの $p_1,\dots,p_n$ でも割り切れない。
    \item<4-> したがって，$q$は新たな素数を約数にもつことになって矛盾する。\qedhere
  \end{enumerate}
\end{proof}
